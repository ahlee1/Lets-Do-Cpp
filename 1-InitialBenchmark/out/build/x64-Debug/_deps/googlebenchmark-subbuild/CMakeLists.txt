                    BINARY_DIR          "C:/Users/Alfred/Documents/Github/Lets-Do-Cpp/1-InitialBenchmark/out/build/x64-Debug/_deps/googlebenchmark-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)
